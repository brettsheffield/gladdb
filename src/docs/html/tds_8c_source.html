35 /* connect to tds database */
36 int db_connect_tds(db_t *db)
37 {
38  LOGINREC *login;
39  DBPROCESS *dbproc;
40  RETCODE erc;
41 
42  if (dbinit() == FAIL) {
43  syslog(LOG_ERR, "dbinit() failed");
44  return -1;
45  }
46 
47  dberrhandle(err_handler);
48  dbmsghandle(msg_handler);
49 
50  if ((login = dblogin()) == NULL) {
51  syslog(LOG_ERR, "unable to allocate tds login structure");
52  return -1;
53  }
54 
55  DBSETLUSER(login, db->user);
56  DBSETLPWD(login, db->pass);
57 
58  if ((dbproc = dbopen(login, db->host)) == NULL) {
59  syslog(LOG_ERR, "unable to connect to %s as %s", db->host,
60  db->user);
61  return -1;
62  }
63 
64  dbloginfree(login);
65 
66  if (db->db && (erc = dbuse(dbproc, db->db)) == FAIL) {
67  syslog(LOG_ERR, "unable to use to database %s", db->db);
68  return -1;
69  }
70 
71  db->conn = dbproc;
72 
73  return 0;
74 }

85 int db_exec_sql_tds(db_t *db, char *sql)
86 {
87  RETCODE erc;
88 
89  /* allocate memory for SQL */
90  if ((erc = dbfcmd(db->conn, "%s ", sql)) == FAIL) {
91  syslog(LOG_ERR, "dbcmd() failed");
92  return -1;
93  }
94 
95  /* ensure we don't have any results pending
96  * - tds won't allow a connection to be reused until
97  * all results have been processed */
98 
99  /* TODO: replace with cancel */
100  int rc;
101  while ((rc = dbresults(db->conn)) == SUCCEED) {
102  while ((rc = dbnextrow(db->conn)) != NO_MORE_ROWS){
103  }
104  }
105 
106  /* execute query */
107  if ((erc = dbsqlexec(db->conn)) == FAIL) {
108  syslog(LOG_ERR, "dbsqlexec() failed");
109  return -1;
110  }
111 
112  return 0;
113 }

115 /* return all results from a SELECT - tds */
116 int db_fetch_all_tds(db_t *db, char *sql, field_t *filter, row_t **rows,
117  int *rowc)
118 {
119 
120  DBPROCESS *dbproc = db->conn;
121  RETCODE erc;
122  field_t *f;
123  field_t *ftmp = NULL;
124  row_t *r;
125  row_t *rtmp = NULL;
126  char *sqltmp;
127 
128  if (filter != NULL) {
129  sqltmp = strdup(sql);
130  asprintf(&sql, "%s WHERE %s='%s'", sqltmp, filter->fname,
131  filter->fvalue);
132  free(sqltmp);
133  }
134 
135  if (db_exec_sql_tds(db, sql) != 0) {
136  return -1;
137  }
138 
139  /* fetch results */
140  while ((erc = dbresults(dbproc)) != NO_MORE_RESULTS) {
141  struct COL
142  {
143  char *name;
144  char *buffer;
145  int type, size, status;
146  } *columns, *pcol;
147 
148  int ncols;
149  int row_code;
150 
151  if (erc == FAIL) {
152  fprintf(stderr, "dbresults failed\n");
153  return -1;
154  }
155  ncols = dbnumcols(dbproc);
156 
157  if ((columns = calloc(ncols, sizeof(struct COL))) == NULL) {
158  perror(NULL);
159  return -1;
160  }
161 
162  /* Read metadata and bind. */
163  for (pcol = columns; pcol - columns < ncols; pcol++) {
164  int c = pcol - columns + 1;
165 
166  pcol->name = dbcolname(dbproc, c);
167  pcol->type = dbcoltype(dbproc, c);
168  pcol->size = dbcollen(dbproc, c);
169 
170  if (SYBCHAR != pcol->type) {
171  pcol->size = dbwillconvert(pcol->type, SYBCHAR);
172  }
173 
174  if ((pcol->buffer = calloc(1, pcol->size + 1)) == NULL){
175  perror(NULL);
176  return -1;
177  }
178 
179  erc = dbbind(dbproc, c, NTBSTRINGBIND,
180  pcol->size+1, (BYTE*)pcol->buffer);
181  if (erc == FAIL) {
182  fprintf(stderr, "dbbind(%d) failed\n", c);
183  return -1;
184  }
185 
186  erc = dbnullbind(dbproc, c, &pcol->status);
187 
188  if (erc == FAIL) {
189  fprintf(stderr, "dbnullbind(%d) failed\n", c);
190  return -1;
191  }
192  }
193  while ((row_code = dbnextrow(dbproc)) != NO_MORE_ROWS){
194  switch (row_code) {
195  case REG_ROW:
196  r = malloc(sizeof(row_t));
197  r->fields = NULL;
198  r->next = NULL;
199  for (pcol=columns; pcol - columns < ncols; pcol++) {
200  f = malloc(sizeof(field_t));
201  f->fname = strdup(pcol->name);
202  asprintf(&f->fvalue, "%s",
203  pcol->status == -1
204  ? "NULL" : pcol->buffer);
205  f->next = NULL;
206  if (r->fields == NULL) {
207  r->fields = f;
208  }
209  if (ftmp != NULL) {
210  ftmp->next = f;
211  }
212  ftmp = f;
213  }
214  if (rtmp == NULL) {
215  /* as this is our first row,
216  * update the ptr */
217  *rows = r;
218  }
219  else {
220  rtmp->next = r;
221  }
222  ftmp = NULL;
223  rtmp = r;
224  break;
225 
226  case BUF_FULL:
227  assert(row_code != BUF_FULL);
228  break;
229 
230  case FAIL:
231  fprintf(stderr, "dbresults failed\n");
232  return -1;
233  break;
234 
235  default:
236  printf("Data for computeid %d ignored\n", row_code);
237  }
238  }
239 
240  /* free metadata and data buffers */
241  for (pcol=columns; pcol - columns < ncols; pcol++) {
242  free(pcol->buffer);
243  }
244  free(columns);
245 
246  /* row count */
247  if (DBCOUNT(dbproc) > -1) (*rowc) = DBCOUNT(dbproc);
248 
249  /* check return status */
250  if (dbhasretstat(dbproc) == TRUE) {
251  syslog(LOG_DEBUG, "Procedure returned %d",
252  dbretstatus(dbproc));
253  }
254  }
255 
256  return 0;
257 }
