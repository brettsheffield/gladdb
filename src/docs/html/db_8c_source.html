315 /** Wrapper to insert SQL into a rdbms
316  * \param db A pointer to a db_t struct
317  * \param resource A pointer to your string containing a resource name
318  * \param data A pointer to a keyval_t containing your data to
319  * insert
320  * \return 0 on success, non-zero (-1) on failure.
321  */
322 int db_insert_sql(db_t *db, char *resource, keyval_t *data)
323 {
324  char *flds = NULL;
325  char *sql;
326  char *vals = NULL;
327  char *tmpflds = NULL;
328  char *tmpvals = NULL;
329  char quot = '\'';
330  int rval;
331  int isconn = 0;
332 
333  dberrcode = NULL;
334  dberror = NULL;
335 
336  /* use backticks to quote mysql */
337  if (strcmp(db->type, "my") == 0)
338  quot = '"';
339 
340  /* build INSERT sql from supplied data */
341  while (data != NULL) {
342  fprintf(stderr, "%s = %s\n", data->key, data->value);
343  if (flds == NULL) {
344  asprintf(&flds, "%s", data->key);
345  asprintf(&vals, "%1$c%2$s%1$c", quot, data->value);
346  }
347  else {
348  tmpflds = strdup(flds);
349  tmpvals = strdup(vals);
350  free(flds); free(vals);
351  asprintf(&flds, "%s,%s", tmpflds, data->key);
352  asprintf(&vals, "%2$s,%1$c%3$s%1$c",
353  quot, tmpvals, data->value);
354  free(tmpflds); free(tmpvals);
355  }
356  data = data->next;
357  }
358 
359  asprintf(&sql,"INSERT INTO %s (%s) VALUES (%s)", resource, flds, vals);
360  free(flds); free(vals);
361  syslog(LOG_DEBUG, "%s", sql);
362 
363  if (db->conn == NULL) {
364  if (db_connect(db) != 0) {
365  syslog(LOG_ERR, "Failed to connect to db on %s",
366  db->host);
367  return -1;
368  }
369  isconn = 1;
370  }
371 
372  rval = db_exec_sql(db, sql);
373  free(sql);
374 
375  /* leave the connection how we found it */
376  if (isconn == 1)
377  db_disconnect(db);
378 
379  return rval;
380 }
