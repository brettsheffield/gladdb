79 
80  res = PQexec(db->conn, sql);
81  int status = PQresultStatus(res);
82  if (status != PGRES_TUPLES_OK && status != PGRES_COMMAND_OK) {
83  dberrcode = PQresultErrorField(res, PG_DIAG_SQLSTATE);
84  dberror = PQerrorMessage(db->conn);
85  syslog(LOG_ERR,

87  PQclear(res);

114 
115  res = PQexec(db->conn, sql);
116  int status = PQresultStatus(res);
117  if (status != PGRES_TUPLES_OK && status != PGRES_COMMAND_OK) {
118  dberrcode = PQresultErrorField(res, PG_DIAG_SQLSTATE);
119  dberror = PQerrorMessage(db->conn);
120  syslog(LOG_ERR, "query failed: %s", dberror);
121  return -1;
122  }
123 
124  /* populate rows and fields */
125  nFields = PQnfields(res);
126  for (j = 0; j < PQntuples(res); j++) {
127  r = malloc(sizeof(row_t));
128  r->fields = NULL;
129  r->next = NULL;
130  for (i = 0; i < nFields; i++) {
131  f = malloc(sizeof(field_t));

133  f->fvalue = strdup(PQgetvalue(res, j, i));
134  f->next = NULL;

136  r->fields = f;

142  }

145  *rows = r;
146  }
147  else {

149  }

151  rtmp = r;
152  }
