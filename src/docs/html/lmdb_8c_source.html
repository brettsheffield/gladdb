43 /* connect to lmdb */
44 int db_connect_lmdb(db_t *db)
45 {
46  int rc;
47  MDB_env *env = NULL;
48 
49  /* Create an LMDB environment handle. */
50  E(mdb_env_create(&env));
51  syslog(LOG_DEBUG, "mdb_env_create() successful in: %s\n",
52  __func__);
53 
54  /* Set the maximum number of threads/reader slots for the environment.
55  * Default is 126, but set to make sure */
56  E(mdb_env_set_maxreaders(env, 126));
57  syslog(LOG_DEBUG, "mdb_env_set_maxreaders() successful in: %s\n",
58  __func__);
59 
60  /* We should set sane defaults as the default db size is small, and allow
61  * args for them too. We will re-grow anyway.
62  * We set the same as what the default is to be sure. */
63  E(mdb_env_set_mapsize(env, 10485760));
64  syslog(LOG_DEBUG, "mdb_env_set_mapsize() successful in: %s\n",
65  __func__);
66 
67  /* MDB_NOSUBDIR is used so db->db is the *actual* name of the db
68  * not the db directory to save the db file in. */
69  E(mdb_env_open(env, db->db, MDB_NOSUBDIR, 0664));
70  syslog(LOG_DEBUG, "mdb_env_open() successful in: %s\n", __func__);
71 
72  /* Save the handle */
73  db->conn = env;
74 
75  return EXIT_SUCCESS;
76 }

98 /* lmdb get */
99 int db_get_lmdb(db_t *db, char *resource, keyval_t *db_data)
100 {
101  int rc;
102  MDB_val key, data;
103 
104  MDB_env *env = db->conn;
105 
106  MDB_txn *txn = NULL;
107  MDB_dbi dbi = NULL;
108 
109  /* Convert our key. */
110  key.mv_size = sizeof(db_data->key);
111  key.mv_data = db_data->key;
112 
113  /* Make sure readonly */
114  E(mdb_txn_begin(env, NULL, MDB_RDONLY, &txn));
115  E(mdb_dbi_open(txn, NULL, 0, &dbi));
116 
117  E(mdb_get(txn, dbi, &key, &data));
118  mdb_txn_abort(txn);
119 
120  /* all good, save data. */
121  db_data->value = data.mv_data;
122 
123  return EXIT_SUCCESS;
124 }

126 /* lmdb put */
127 int db_insert_lmdb(db_t *db, char *resource, keyval_t *db_data)
128 {
129  /* We don't use *resource yet, we can support multiple databases or nested transactions,
130  * so that's something to consider later. */
131 
132  /* Maintain our stats */
133  MDB_stat mst;
134 
135  /* We need to convert/set keyval_t to MDB_val */
136  MDB_val key, data;
137 
138  key.mv_size = sizeof(db_data->key);
139  key.mv_data = db_data->key;
140 
141  data.mv_size = sizeof(db_data->value);
142  data.mv_data = db_data->value;
143 
144  /* Get ready */
145  int rc;
146  MDB_txn *txn = NULL;
147  MDB_dbi dbi;
148  MDB_env *env = db->conn;
149 
150  /* Once the environment is open via db_connect_lmdb(), a transaction can be created within it
151  * using mdb_txn_begin(). Transactions may be read-write or read-only, and read-write
152  * transactions may be nested. We always read-write here.
153  *
154  * A transaction must only be used by one thread at a time. Transactions are always required,
155  * even for read-only access. The transaction provides a consistent view of the
156  * data. */
157  E(mdb_txn_begin(db->conn, NULL, 0, &txn)); /* We don't support nested transactions yet. */
158  E(mdb_dbi_open(txn, NULL, 0, &dbi));
159 
160  /* We don't do anything complicated yet or allow storing multiple values for a key.
161  * We could use *resource as a flag to indicate that and switch to mdb_cursor_put().
162  * We also don't use and of the flexible flags. See mdb_put() docs. */
163  E(mdb_put(txn, dbi, &key, &data, 0));
164  E(mdb_txn_commit(txn));
165  E(mdb_env_stat(env, &mst));
166 
167  return EXIT_SUCCESS;
168 }

172 {

175 
176  int rc;
177  MDB_txn *txn = NULL;
178  MDB_val key;
179  MDB_dbi dbi = NULL;
180  MDB_env *env = db->conn;
181 
182  /* Convert */
183  key.mv_size = sizeof(db_data->key);
184  key.mv_data = db_data->key;
185 
186  E(mdb_txn_begin(db->conn, NULL, 0, &txn)); /* We don't support nested transactions yet. */
187  E(mdb_dbi_open(txn, NULL, 0, &dbi));
188 
189  /* Again, basic delete, no flags. As with db_insert_mdb() we could use
190  * *resource and switch to mdb_cursor_del() */
191  if (RES(MDB_NOTFOUND, mdb_del(txn, dbi, &key, NULL))) {
192  mdb_txn_abort(txn);
193  } else {
194  E(mdb_txn_commit(txn));
195  }
196  E(mdb_env_stat(env, &mst));
197 
198  return EXIT_SUCCESS;
199 }
