32 int db_connect_ldap(db_t *db)
33 {
34  LDAP *l;
35  int rc;
36  int protocol = LDAP_VERSION3;
37 
38  rc = ldap_initialize(&l, db->host);
39  if (rc != LDAP_SUCCESS) {
40  syslog(LOG_DEBUG,
41  "Could not create LDAP session handle for URI=%s (%d): %s\n",
42  db->host, rc, ldap_err2string(rc));
43  return -1;
44  }
45  syslog(LOG_DEBUG, "ldap_initialise() successful");
46 
47  /* Ensure we use LDAP v3 */
48  rc = ldap_set_option(l, LDAP_OPT_PROTOCOL_VERSION, &protocol);
49 
50  db->conn = l;
51 
52  return 0;
53 }

72 int db_fetch_all_ldap(db_t *db, char *query, field_t *filter, row_t **rows,
73  int *rowc)
74 {
75  BerElement *ber;
76  LDAPMessage *msg;
77  LDAPMessage *res = NULL;
78  char *a;
79  char *lfilter = NULL;
80  char *search;
81  char **vals;
82  int i;
83  int rc;
84  field_t *f;
85  field_t *ftmp = NULL;
86  row_t *r;
87  row_t *rtmp = NULL;
88 
89  asprintf(&search, "%s,%s", query, db->db);
90  rc = ldap_search_ext_s(db->conn, search, LDAP_SCOPE_SUBTREE,
91  lfilter, NULL, 0, NULL, NULL, LDAP_NO_LIMIT,
92  LDAP_NO_LIMIT, &res);
93  free(search);
94 
95  if (rc != LDAP_SUCCESS) {
96  syslog(LOG_DEBUG, "search error: %s (%d)",
97  ldap_err2string(rc), rc);
98  return -1;
99  }
100  syslog(LOG_DEBUG, "ldap_search_ext_s successful");
101 
102  *rowc = ldap_count_messages(db->conn, res);
103  syslog(LOG_DEBUG, "Messages: %i", *rowc);
104 
105  /* populate rows and fields */
106  for (msg = ldap_first_entry(db->conn, res); msg != NULL;
107  msg = ldap_next_entry(db->conn, msg))
108  {
109  r = malloc(sizeof(row_t));
110  r->fields = NULL;
111  r->next = NULL;
112  for (a = ldap_first_attribute(db->conn, msg, &ber); a != NULL;
113  a = ldap_next_attribute(db->conn, msg, ber))
114  {
115  /* attributes may have more than one value - here
116  we list them all as separate fields */
117  if ((vals = ldap_get_values(db->conn, msg, a)) != NULL)
118  {
119  for (i = 0; vals[i] != NULL; i++) {
120  f = malloc(sizeof(field_t));
121  f->fname = strdup(a);
122  f->fvalue = strdup(vals[i]);
123  f->next = NULL;
124  if (r->fields == NULL) {
125  r->fields = f;
126  }
127  if (ftmp != NULL) {
128  ftmp->next = f;
129  }
130  ftmp = f;
131  }
132  ldap_value_free(vals);
133  }
134  ldap_memfree(a);
135  }
136  ldap_memfree(ber);
137  if (rtmp == NULL) {
138  /* as this is our first row, update the ptr */
139  *rows = r;
140  }
141  else {
142  rtmp->next = r;
143  }
144  ftmp = NULL;
145  rtmp = r;
146  }
147  ldap_msgfree(res);
148  return 0;
149 }

151 /* ldap add */
152 int db_insert_ldap(db_t *db, char *resource, keyval_t *data)
153 {
154  LDAPMod **lmod;
155  int rc;
156  char *dn;
157 
158  /* build dn from first key, view and ldap base */
159  asprintf(&dn, "%s=%s,%s,%s", data->key, data->value, resource, db->db);
160 
161  rc = keyval_to_LDAPMod(data, &lmod);
162  if (rc == 0) {
163  ldap_mods_free(lmod, 1);
164  return 1;
165  }
166  rc = db_connect_ldap(db);
167  if (rc != 0) goto db_insert_ldap_cleanup;
168  rc = ldap_simple_bind_s(db->conn, db->user, db->pass);
169  if (rc != 0) goto db_insert_ldap_cleanup;
170  rc = ldap_add_s(db->conn, dn, lmod);
171 
172 db_insert_ldap_cleanup:
173  db_disconnect_ldap(db);
174  ldap_mods_free(lmod, 1);
175  free(dn);
176  return (rc == LDAP_SUCCESS) ? 0 : 1;
177 }

179 /* take struct keyval_t, bundle up duplicate attributes and spit
180  out an LDAPMod array
181 */
182 int keyval_to_LDAPMod(keyval_t *kv, LDAPMod ***lm)
183 {
184  if (kv == NULL) return 1;
185  keyval_t *k = kv;
186  char *last = kv->key;
187  char **vals;
188  int c = 0;
189  int i = 0;
190  int total = 0;
191  int unique = 0;
192 
193  count_keyvals(kv, &total, &unique);
194  *lm = calloc(total, sizeof(LDAPMod));
195  vals = calloc(total, sizeof(char *));
196  k = kv;
197  while (k != NULL) {
198  if (strcmp(last, k->key) != 0 || k->next == NULL) {
199  /* we have all of our values */
200  if (k->next == NULL && strcmp(last, k->key) == 0) {
201  vals[i++] = strdup(k->value);
202  }
203  (*lm)[c] = calloc(1, sizeof(LDAPMod));
204  (*lm)[c]->mod_type = strdup(last);
205  (*lm)[c]->mod_values = vals;
206  if (k->next != NULL) {
207  vals = calloc(total, sizeof(char *));
208  }
209  else {
210  if (strcmp(last, k->key) != 0) {
211  c++;
212  (*lm)[c] = calloc(1, sizeof(LDAPMod));
213  (*lm)[c]->mod_type = strdup(k->key);
214  vals = calloc(2, sizeof(char *));
215  vals[0] = strdup(k->value);
216  (*lm)[c]->mod_values = vals;
217  break;
218  }
219  }
220  last = k->key;
221  c++;
222  i = 0;
223  }
224  if (k->next != NULL) vals[i++] = strdup(k->value);
225  k = k->next;
226  }
227 
228  return unique;
229 }

231 /* return 0 if LDAPMods match, else 1 */
232 int compare_LDAPMod(LDAPMod **lm0, LDAPMod **lm1)
233 {
234  int i, j;
235  int rc = 0;
236  char *t0, *t1, **v0, **v1;
237  const int depth = 65535;
238 
239  if (lm0 == NULL && lm1 == NULL) return 0;
240  if (lm0 == NULL || lm1 == NULL) return 1;
241 
242  for (i = 0; i < depth; i++) {
243  if (lm0[i] == NULL || lm1[i] == NULL) break;
244  t0 = lm0[i]->mod_type;
245  t1 = lm1[i]->mod_type;
246  v0 = lm0[i]->mod_values;
247  v1 = lm1[i]->mod_values;
248  if (t0 == NULL && t1 == NULL) break;
249  if (t0 == NULL || t1 == NULL) return 1;
250  for (j = 0; j < depth; j++) {
251  if (strcmp(t0, t1) != 0) return 1;
252  if (v0[j] == NULL && v1[j] == NULL) break;
253  if (v0[j] == NULL || v1[j] == NULL) return 1;
254  if (strcmp(v0[j], v1[j]) != 0) return 1;
255  }
256  if (j == depth) return 1;
257  }
258  if (i == depth) return 1;
259 
260  return rc;
261 }
