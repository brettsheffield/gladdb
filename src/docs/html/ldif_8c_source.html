52  while (rc == 0) {

54  if (rc == 0) {
55  syslog(LOG_DEBUG, "EOF reached for ldif data");
56  return 1;
57  }
58  else if (rc == -1) {
59  syslog(LOG_ERR, "error reading ldif record");
60  return 0;
61  }
62  rbuf.bv_val = bufp;
63  rbuf.bv_len = 0;
64  rc = ldap_parse_ldif_record(&rbuf, linenum, &lr, "libgladdb",
65  flags);
66  if (rc != 0) {
67  syslog(LOG_DEBUG, "failed to parse ldif record");
68  return 0;
69  }
70  if (lr.lr_op == LDAP_REQ_DELETE) {
71  syslog(LOG_DEBUG, "DELETE resulting from ldif record");
72  rc = ldap_delete_ext_s(db->conn, lr.lr_dn.bv_val,
73  lr.lr_ctrls, NULL);
74  }
75  else if (lr.lr_op == LDAP_REQ_RENAME) {
76  syslog(LOG_DEBUG, "RENAME resulting from ldif record");
77  char *newparent = NULL;
78  if (lr.lrop_newsup.bv_val) {
79  newparent = lr.lrop_newsup.bv_val;
80  }
81  rc = ldap_rename_s(db->conn, lr.lr_dn.bv_val,

83  lr.lrop_delold, lr.lr_ctrls, NULL);
84  }
85  else if (lr.lr_op == LDAP_REQ_ADD) {
86  syslog(LOG_DEBUG, "ADD resulting from ldif record");
